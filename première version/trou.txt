<script>
    function toggle(e) {
        var box = document.getElementById(e)
        var content = box.parentNode
        var barHider = content.previousElementSibling

        if (content.style.display == "none") {
            barHider.style.display = "none"
            content.style.display = "flex";
        }
        else {
            barHider.style.display = "block"
            content.style.display = "none"
        }
    }

    var tagsDiv = document.getElementById("tags");
    var flagsDiv = document.getElementById("flags");

    var allSections = document.getElementsByClassName("section");
    var allHiders = document.getElementsByClassName("barHider");
    var allBars = document.getElementsByClassName("bar");

    if (tagsDiv.innerHTML.includes("Difficile") || flagsDiv.innerHTML.includes("flag2")) {
        for (var sectionDiv of allSections) {
            sectionDiv.style.borderColor = "red red red red";
        }

        for (var hiderDiv of allHiders) {
            hiderDiv.style.borderColor = "red red red red";
        }

        for (var barDiv of allBars) {
            barDiv.style.borderColor = "red red red red";
        }
    }

</script>
